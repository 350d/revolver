Keyboard Navigation Example

In this example, there are no controls to click on, just a simple example to show you how easy it would be to add keyboard navigation to your slider.

Source

var revolver = $('#my_slider').revolver({transition: {type: 'slide'}}).data('revolver');

$(document).keypress(function(e)
{
    // right
    if (e.keyCode === 39)
    {
        revolver.next();
    }

    // left
    else if (e.keyCode === 37)
    {
        revolver.previous({direction:'right'});
    }

    // up
    else if (e.keyCode === 38)
    {
        revolver.previous({direction:'up'});
    }

    // down
    else if (e.keyCode === 40)
    {
        revolver.next({direction:'down'});
    }
});
