Keyboard Navigation Example

In this example, there are no controls to click on, just a simple example to show you how easy it would be to add keyboard navigation to your slider. Try pressing the left, right, up, and down arrows on your keyboard.

#my_slider {
    width: 720px;
    height: 300px;
    overflow: hidden;
    position: relative;
}

#my_slider .slide {
    left: 0;
    position: absolute;
    top: 0;
}

#my_slider .hidden {
    display: none;
}
var revolver = $('#my_slider').revolver({transition: {type: 'slide'}}).data('revolver');

$(document).keydown(function(e)
{
    // right
    if (e.keyCode === 39)
    {
        revolver.next();
    }

    // left
    else if (e.keyCode === 37)
    {
        revolver.previous({direction:'right'});
    }

    // up
    else if (e.keyCode === 38)
    {
        e.preventDefault();
        revolver.next({direction:'up'});
    }

    // down
    else if (e.keyCode === 40)
    {
        e.preventDefault();
        revolver.previous({direction:'down'});
    }
});
var revolver = $('my_slider').revolver({transition: {type: 'slide'}}).data('revolver');

$(document).keydown(function(e)
{
    // right
    if (e.key === "right")
    {
        revolver.next();
    }

    // left
    else if (e.key === "left")
    {
        revolver.previous({direction:'right'});
    }

    // up
    else if (e.key === "up")
    {
        e.preventDefault();
        revolver.next({direction:'up'});
    }

    // down
    else if (e.key === "down")
    {
        e.preventDefault();
        revolver.previous({direction:'down'});
    }
});

Now, most often, you'll either use left and right or up and down, not both. Unless you do something really creative to make it cool.

You could also use this same principle to add keyboard shortcuts for pause, play, stop, etc.
